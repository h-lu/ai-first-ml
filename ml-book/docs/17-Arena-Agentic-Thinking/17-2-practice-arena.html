#| eval: false
#| code-fold: true
#| code-summary: "点击查看Gradio竞技场的完整代码"

import gradio as gr
import torch
from transformers import AutoModelForCausalLM, AutoTokenizer
import random

# --- 1. 后端逻辑：加载模型并生成回答 ---

# 加载我们训练好的两个模型
sft_model_path = "./sft_bean_voyage_output/final"
dpo_model_path = "./dpo_bean_voyage_output/final"

tokenizer = AutoTokenizer.from_pretrained(sft_model_path)
sft_model = AutoModelForCausalLM.from_pretrained(sft_model_path, torch_dtype=torch.bfloat16, device_map="auto")
dpo_model = AutoModelForCausalLM.from_pretrained(dpo_model_path, torch_dtype=torch.bfloat16, device_map="auto")

print("SFT和DPO模型已加载完毕。")

# 核心函数：获取两个模型的回答并随机排序
def get_responses(prompt):
    print(f"收到问题: {prompt}")
    
    # 统一使用DPO模型的Tokenizer进行编码
    inputs = tokenizer(prompt, return_tensors="pt").to(sft_model.device)

    # 分别生成回答
    with torch.no_grad():
        sft_output = sft_model.generate(**inputs, max_new_tokens=100)
        dpo_output = dpo_model.generate(**inputs, max_new_tokens=100)

    sft_response = tokenizer.decode(sft_output[0], skip_special_tokens=True)
    dpo_response = tokenizer.decode(dpo_output[0], skip_special_tokens=True)
    
    print(f"SFT模型回答: {sft_response}")
    print(f"DPO模型回答: {dpo_response}")

    # 随机打乱顺序以实现盲测
    responses = [sft_response, dpo_response]
    random.shuffle(responses)
    
    return responses[0], responses[1]

# --- 2. 前端界面：用Gradio搭建 ---

with gr.Blocks(theme=gr.themes.Base(), title="咖啡豆AI竞技场") as arena:
    gr.Markdown("# 咖啡豆奇旅AI竞技场\n\n输入你的问题，然后为匿名模型的回答投票！")
    
    with gr.Row():
        with gr.Column():
            prompt_textbox = gr.Textbox(label="输入你的问题", lines=4)
            generate_btn = gr.Button("生成回答", variant="primary")
            
        with gr.Column():
            response_A = gr.Textbox(label="模型 A 回答", lines=4, interactive=False)
            response_B = gr.Textbox(label="模型 B 回答", lines=4, interactive=False)

    with gr.Row():
        vote_A_btn = gr.Button("模型A更好")
        vote_B_btn = gr.Button("模型B更好")
        tie_btn = gr.Button("平局")

    # 定义按钮的点击事件
    generate_btn.click(
        fn=get_responses,
        inputs=[prompt_textbox],
        outputs=[response_A, response_B]
    )
    
    # (在真实应用中，投票按钮会连接到记录结果的后端逻辑)
    vote_A_btn.click(lambda: gr.Info("感谢您的投票！已记录：模型A获胜。"))
    vote_B_btn.click(lambda: gr.Info("感谢您的投票！已记录：模型B获胜。"))
    tie_btn.click(lambda: gr.Info("感谢您的投票！已记录：平局。"))


print("正在启动Gradio竞技场...")
arena.launch(share=True) # share=True会生成一个公开链接
